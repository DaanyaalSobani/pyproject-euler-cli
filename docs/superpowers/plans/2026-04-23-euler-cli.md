> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement login, logout, session persistence, answer submission, and account status for euler-cli — a terminal tool that authenticates to projecteuler.net via HTTP form login and caches the session cookie locally.

**Architecture:** Thin Click CLI layer (`cli.py`) delegates to focused modules. `config.py` holds constants and loads dev.env credentials. `session.py` manages the cookie file at `~/.euler/session.json`. `auth.py` handles login/logout. `submit.py` submits answers. `status.py` fetches account info. All HTTP calls go through `requests.Session`; HTML parsing uses BeautifulSoup.

**Tech Stack:** Python 3.10+, click 8, requests, beautifulsoup4, keyring, rich, pytest, unittest.mock

> **Browser access:** Chrome DevTools MCP and Playwright MCP are already configured in this environment. Task 4 uses `mcp__plugin_chrome-devtools-mcp_chrome-devtools__*` or `mcp__plugin_playwright_playwright__*` tools directly.

Terminal CLI to submit Project Euler answers without leaving your editor. Authenticates to projecteuler.net via standard HTML form login.

```python
from pathlib import Path
import keyring as _keyring

BASE_URL = "https://projecteuler.net"
SESSION_FILE = Path.home() / ".euler" / "session.json"
KEYRING_SERVICE = "euler-cli"
KEYRING_USERNAME_KEY = "__euler_username__"
DEV_ENV_PATH = Path(__file__).parent.parent / "dev.env"


def load_dev_env() -> dict[str, str]:
    if not DEV_ENV_PATH.exists():
        return {}
    result: dict[str, str] = {}
    for line in DEV_ENV_PATH.read_text().splitlines():
        line = line.strip()
        if line and not line.startswith("#") and "=" in line:
            key, _, value = line.partition("=")
            result[key.strip()] = value.strip()
    return result


def get_credentials() -> tuple[str, str] | None:
    dev = load_dev_env()
    if "EULER_USERNAME" in dev and "EULER_PASSWORD" in dev:
        return dev["EULER_USERNAME"], dev["EULER_PASSWORD"]
    username = _keyring.get_password(KEYRING_SERVICE, KEYRING_USERNAME_KEY)
    if not username:
        return None
    password = _keyring.get_password(KEYRING_SERVICE, username)
    return (username, password) if password else None


def save_credentials(username: str, password: str) -> None:
    _keyring.set_password(KEYRING_SERVICE, KEYRING_USERNAME_KEY, username)
    _keyring.set_password(KEYRING_SERVICE, username, password)


def delete_credentials() -> None:
    username = _keyring.get_password(KEYRING_SERVICE, KEYRING_USERNAME_KEY)
    if username:
        for key in (username, KEYRING_USERNAME_KEY):
            try:
                _keyring.delete_password(KEYRING_SERVICE, key)
            except Exception:
                pass
```

```python
import json
import requests
from . import config


def load_session() -> requests.Session | None:
    if not config.SESSION_FILE.exists():
        return None
    try:
        cookies = json.loads(config.SESSION_FILE.read_text())
        s = requests.Session()
        for c in cookies:
            s.cookies.set(
                c["name"], c["value"],
                domain=c.get("domain", ""),
                path=c.get("path", "/"),
            )
        return s
    except (json.JSONDecodeError, KeyError, TypeError):
        return None


def save_session(s: requests.Session) -> None:
    config.SESSION_FILE.parent.mkdir(parents=True, exist_ok=True)
    cookies = [
        {"name": c.name, "value": c.value, "domain": c.domain, "path": c.path}
        for c in s.cookies
    ]
    config.SESSION_FILE.write_text(json.dumps(cookies))


def clear_session() -> None:
    config.SESSION_FILE.unlink(missing_ok=True)
```

```python
import requests
from bs4 import BeautifulSoup
from . import config, session

# Field names — update from docs/site-notes.md after Task 4
_CSRF_FIELD = "form_token"
_USERNAME_FIELD = "username"
_PASSWORD_FIELD = "password"
_SUBMIT_FIELD = "sign_in"

_LOGIN_URL = f"{config.BASE_URL}/sign_in"


def _extract_csrf(html: str) -> str:
    soup = BeautifulSoup(html, "html.parser")
    token = soup.find("input", {"name": _CSRF_FIELD})
    if token is None:
        raise ValueError(f"CSRF field '{_CSRF_FIELD}' not found on page")
    return token["value"]


def _extract_error(html: str) -> str | None:
    soup = BeautifulSoup(html, "html.parser")
    el = soup.find(class_="error") or soup.find("span", class_="error")
    return el.get_text(strip=True) if el else None


def _extract_display_name(html: str) -> str | None:
    # Update selector from docs/site-notes.md
    soup = BeautifulSoup(html, "html.parser")
    for attrs in ({"href": "/account"}, {"id": "lnkAccount"}, {"class": "username"}):
        el = soup.find("a", attrs)
        if el:
            return el.get_text(strip=True)
    return None


def login(username: str, password: str) -> str:
    """Authenticate to projecteuler.net. Returns display name. Raises ValueError on failure."""
    s = requests.Session()
    resp = s.get(_LOGIN_URL)
    resp.raise_for_status()

    token = _extract_csrf(resp.text)
    post_resp = s.post(_LOGIN_URL, data={
        _USERNAME_FIELD: username,
        _PASSWORD_FIELD: password,
        _SUBMIT_FIELD: "Sign In",
        _CSRF_FIELD: token,
    })
    post_resp.raise_for_status()

    error = _extract_error(post_resp.text)
    if error:
        raise ValueError(error)

    display_name = _extract_display_name(post_resp.text)
    if display_name is None:
        raise ValueError("Login failed: could not detect logged-in state in response")

    session.save_session(s)
    return display_name


def logout(remove_keyring: bool = False) -> None:
    session.clear_session()
    if remove_keyring:
        config.delete_credentials()
```

@main.command()

    creds = config.get_credentials()

        config.save_credentials(username, password)

from bs4 import BeautifulSoup
from . import config, session

# Field names — update from docs/site-notes.md after Task 4
_FORM_ID = "problem_answer"
_ANSWER_FIELD = "guess"
_CSRF_FIELD = "form_token"


def _extract_form(html: str) -> tuple[str, str, str]:
    """Returns (post_url, answer_field_name, csrf_token)."""
    soup = BeautifulSoup(html, "html.parser")
    form = soup.find("form", {"id": _FORM_ID})
    if form is None:
        raise ValueError(f"Answer form #{_FORM_ID} not found on page")
    action = form.get("action", "")
    answer_input = form.find("input", {"name": _ANSWER_FIELD})
    field_name = answer_input["name"] if answer_input else _ANSWER_FIELD
    csrf = form.find("input", {"name": _CSRF_FIELD})
    token = csrf["value"] if csrf else ""
    return f"{config.BASE_URL}/{action}", field_name, token


def _is_correct(html: str) -> bool:
    # Update phrase from docs/site-notes.md after Task 4
    return "that is the correct answer" in html.lower()


def submit_answer(problem: int, answer: str) -> bool:
    """Submit answer to problem N. Returns True if correct. Raises PermissionError if not logged in."""
    s = session.load_session()
    if s is None:
        raise PermissionError("Not logged in")

    resp = s.get(f"{config.BASE_URL}/problem={problem}")
    resp.raise_for_status()

    if "sign_in" in resp.url:
        raise PermissionError("Session expired")

    post_url, field_name, token = _extract_form(resp.text)
    post_resp = s.post(post_url, data={field_name: answer, _CSRF_FIELD: token})
    post_resp.raise_for_status()
    return _is_correct(post_resp.text)

_PROGRESS_PATH = "/progress"


def get_status() -> dict:
    """Returns dict with 'username' (str) and 'solved' (int). Raises PermissionError if not logged in."""
    s = session.load_session()
    if s is None:
        raise PermissionError("Not logged in")

    resp = s.get(f"{config.BASE_URL}{_PROGRESS_PATH}")
    resp.raise_for_status()

    if "sign_in" in resp.url:
        raise PermissionError("Session expired")

    soup = BeautifulSoup(resp.text, "html.parser")

    # Username — update selector from docs/site-notes.md
    username = "Unknown"
    for attrs in ({"class": "username"}, {"id": "lnkAccount"}, {"href": "/account"}):
        el = soup.find(attrs=attrs)
        if el:
            username = el.get_text(strip=True)
            break

    # Solve count — update selector from docs/site-notes.md
    solved = 0
    progress_el = soup.find(id="progress_info")
    if progress_el:
        match = re.search(r"(\d+)", progress_el.get_text())
        if match:
            solved = int(match.group(1))

    return {"username": username, "solved": solved}
